#version 330 core

layout (location = 1) in vec2 inTexCoord;
layout (location = 0) in vec3 position;

out vec2 texCoord;

uniform mat4 world;
uniform	mat4 view;
uniform	mat4 projection;

void main()
{
	texCoord = inTexCoord;
	vec4 clipSpacePosition = vec4(position.xyz, 1) * world  * view * projection;
	gl_Position = clipSpacePosition;
}
